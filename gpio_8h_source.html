461 static inline uint32_t gpio_get_irq_event_mask(uint gpio) {
462  check_gpio_param(gpio);
463  io_irq_ctrl_hw_t *irq_ctrl_base = get_core_num() ?
464  &iobank0_hw->proc1_irq_ctrl : &iobank0_hw->proc0_irq_ctrl;
